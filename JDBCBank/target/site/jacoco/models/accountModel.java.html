package models;

public class accountModel {

	}

		this.name = name;

	}
	public void setUser(int user) {
		this.user = user;
	}
	public double getBalance() {
		return balance;
	}
	public void setBalance(double balance) {
		this.balance = balance;
	}
	private double balance;
}
